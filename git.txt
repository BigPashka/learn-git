git init	инициализация репозитория
git rm --cached <имя файла>	перестать отслеживать файл
git commit -m "<имя комита/сообщение>"
git branch 			список веток
git branch <имя ветки>		создать ветку с определённым именем
git branch -d <имя ветки>	удалить ветку
git checkout <имя ветки>	переключиться на другую ветку
git checkout -b <имя ветки>	создать ветку и переключиться на неё
git merge <имя ветки>		совместить текущую ветку с другой
git push	загрузить в гитхаб
git pull	выгрузить из хаба
